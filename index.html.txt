function renderizarPreguntasFisicas() {
    const contP2P5 = document.getElementById('contenedor-p2p5');
    const contP6 = document.getElementById('contenedor-p6');
    const contP7 = document.getElementById('contenedor-p7');
    
    contP2P5.innerHTML = ""; contP6.innerHTML = ""; contP7.innerHTML = "";

    ["P2", "P3", "P4", "P5"].forEach(elem => {
        contP2P5.innerHTML += `<div class="ct">${ESTRUCTURA_NORMA[elem].titulo}</div>`;
        ESTRUCTURA_NORMA[elem].preguntas.forEach(p => { contP2P5.innerHTML += crearHTMLBloquePregunta(p); });
    });

    ["P6"].forEach(elem => {
        contP6.innerHTML += `<div class="ct">${ESTRUCTURA_NORMA[elem].titulo} (26 puntos oficiales)</div>`;
        ESTRUCTURA_NORMA[elem].preguntas.forEach(p => { contP6.innerHTML += crearHTMLBloquePregunta(p); });
    });

    ["P7"].forEach(elem => {
        contP7.innerHTML += `<div class="ct">${ESTRUCTURA_NORMA[elem].titulo}</div>`;
        ESTRUCTURA_NORMA[elem].preguntas.forEach(p => { contP7.innerHTML += crearHTMLBloquePregunta(p); });
    });
}

function switchPage(id, tabElement) {
    document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
    document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
    document.getElementById('page-' + id).classList.add('active');
    tabElement.classList.add('active');
}

function toggleGuia(id) {
    const el = document.getElementById(id);
    el.style.display = (el.style.display === 'block') ? 'none' : 'block';
}

function crearNuevaAuditoria() {
    const id = "VDA-" + Date.now().toString().slice(-6);
    DB.auditorias[id] = {
        id: id, auditor: "", proveedor: "", fecha: new Date().toISOString().slice(0,10), proceso: "",
        scores: {}, findings: {}
    };
    DB.activa = id;
    guardarEnLocalStorage();
    cargarAuditoriaEnPantalla(id);
    actualizarListaAuditorias();
    switchPage('cabecera', document.querySelectorAll('.tab')[1]);
}

function cargarAuditoriaEnPantalla(id) {
    DB.activa = id;
    const a = DB.auditorias[id];
    document.getElementById('lbl-app-title').textContent = "Auditoría: " + id;
    
    document.getElementById('audit_id_field').value = a.id;
    document.getElementById('auditor_name').value = a.auditor || "";
    document.getElementById('comercial-name').value = a.proveedor || "";
    document.getElementById('audit_date').value = a.fecha || "";
    document.getElementById('proceso_auditado').value = a.proceso || "";

    document.querySelectorAll('.score-opt').forEach(b => b.classList.remove('sel'));
    document.querySelectorAll('textarea').forEach(t => t.value = "");

    Object.keys(a.scores).forEach(qid => {
        const val = a.scores[qid];
        const cont = document.querySelector(`[data-qid="${qid}"]`);
        if(cont) {
            cont.querySelectorAll('.score-opt').forEach(btn => {
                if(btn.textContent == val) btn.classList.add('sel');
            });
        }
    });

    Object.keys(a.findings).forEach(qid => {
        const txt = document.getElementById(`find_${qid}`);
        if(txt) txt.value = a.findings[qid];
    });

    calcularResultadosVDA();
}

function seleccionarNotaMóvil(qid, nota, boton) {
    if(!DB.activa) return;
    const cont = boton.parentElement;
    cont.querySelectorAll('.score-opt').forEach(b => b.classList.remove('sel'));
    boton.classList.add('sel');
    
    DB.auditorias[DB.activa].scores[qid] = nota;
    guardarEnLocalStorage();
    calcularResultadosVDA();
}

function dictarVozMóvil(qid, boton) {
    if(!DB.activa) return;
    const Speech = window.SpeechRecognition || window.webkitSpeechRecognition;
    if(!Speech) {
        alert("El dictado por voz no está disponible en este navegador móvil.");
        return;
    }

    const rec = new Speech();
    rec.lang = 'es-ES';
    rec.interimResults = false;
    boton.classList.add('listening');
    rec.start();

    rec.onresult = function(e) {
        const frase = e.results[0][0].transcript;
        const txtEl = document.getElementById(`find_${qid}`);
        if(txtEl.value.trim() === "") {
            txtEl.value = frase;
        } else {
            const ultimoChar = txtEl.value.trim().slice(-1);
            const separador = [".", ",", "!", "?"].includes(ultimoChar) ? " " : ". ";
            txtEl.value = txtEl.value.trim() + separador + frase;
        }
        DB.auditorias[DB.activa].findings[qid] = txtEl.value;
        guardarEnLocalStorage();
    };

    rec.onspeechend = function() { rec.stop(); boton.classList.remove('listening'); };
    rec.onerror = function() { boton.classList.remove('listening'); };
}

function calcularResultadosVDA() {
    if(!DB.activa) return;
    const a = DB.auditorias[DB.activa];
    
    let sumTotal = 0, countTotal = 0;
    let desgloses = {};

    Object.keys(ESTRUCTURA_NORMA).forEach(elem => {
        let sumElem = 0, countElem = 0;
        ESTRUCTURA_NORMA[elem].preguntas.forEach(p => {
            if(a.scores[p.id] !== undefined && a.scores[p.id] !== "") {
                const nota = parseInt(a.scores[p.id]);
                sumElem += nota;
                countElem += 1;
                sumTotal += nota;
                countTotal += 1;
            }
        });
        if(countElem > 0) {
            desgloses[elem] = Math.round((sumElem / (countElem * 10)) * 100);
        }
    });

    if(countTotal === 0) {
        document.getElementById('res-total').textContent = "0%";
        document.getElementById('res-clase').textContent = "-";
        document.getElementById('res-desglose').innerHTML = "No hay entradas evaluadas.";
        return;
    }

    const eTotal = Math.round((sumTotal / (countTotal * 10)) * 100);
    document.getElementById('res-total').textContent = eTotal + "%";

    let clasificacion = "C";
    if(eTotal >= 90) clasificacion = "A";
    else if(eTotal >= 80) clasificacion = "B";

    document.getElementById('res-clase').textContent = clasificacion;

    let htmlDesglose = "";
    Object.keys(desgloses).forEach(el => {
        htmlDesglose += `<div><strong>Elemento ${el}:</strong> ${desgloses[el]}% de cumplimiento</div>`;
    });
    document.getElementById('res-desglose').innerHTML = htmlDesglose;
}

function guardarEnLocalStorage() {
    localStorage.setItem('vda63_master_db_v2', JSON.stringify(DB));
    const toast = document.getElementById('save-toast');
    toast.style.display = 'block';
    setTimeout(() => { toast.style.display = 'none'; }, 400);
}

function cargarTodoDeLocalStorage() {
    const raw = localStorage.getItem('vda63_master_db_v2');
    if(raw) {
        try {
            DB = JSON.parse(raw);
            if(DB.activa && DB.auditorias[DB.activa]) {
                cargarAuditoriaEnPantalla(DB.activa);
            }
        } catch(e) { console.error(e); }
    }
}

function actualizarListaAuditorias() {
    const cont = document.getElementById('lista-auditorias-container');
    cont.innerHTML = "";
    if(Object.keys(DB.auditorias).length === 0) {
        cont.innerHTML = "<p style='font-size:13px; color:var(--text3); text-align:center; padding:20px;'>No hay auditorías guardadas en este dispositivo.</p>";
        return;
    }
    Object.keys(DB.auditorias).forEach(id => {
        const aud = DB.auditorias[id];
        const activoStyle = (id === DB.activa) ? `border-color: var(--accent);` : '';
        cont.innerHTML += `
        <div class="card" style="${activoStyle}" onclick="cargarAuditoriaEnPantalla('${id}')">
            <div style="display:flex; justify-content:space-between; align-items:center;">
                <strong>📂 ${id}</strong> 
                <span style="font-size:11px; color:var(--accent);">${aud.fecha}</span>
            </div>
            <div style="font-size:13px; color:var(--text2); margin-top:4px;">Proveedor: ${aud.proveedor || 'No asignado'}</div>
        </div>`;
    });
}

function eliminarAuditoriaActual() {
    if(!DB.activa) return;
    if(confirm("¿Seguro que deseas borrar permanentemente este registro?")) {
        delete DB.auditorias[DB.activa];
        DB.activa = Object.keys(DB.auditorias)[0] || null;
        guardarEnLocalStorage();
        if(DB.activa) cargarAuditoriaEnPantalla(DB.activa);
        else location.reload();
    }
}
